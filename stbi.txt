    import pandas as pd
    import xml.etree.ElementTree as ET

    tree = ET.parse('discogs_releases.xml')
    root = tree.getroot()

    data = []
    for release in root.findall('release'):
        title = release.find('title').text if release.find('title') is not None else ''
        artist = release.find('artists/artist/name').text if release.find('artists/artist/name') is not None else ''
        year = release.find('year').text if release.find('year') is not None else ''
        data.append([title, artist, year])

    df = pd.DataFrame(data, columns=['title', 'artist', 'year'])
    df.to_csv('discogs_releases.csv', index=False)
